@nukedTime = 7200

terraform_link = {
    to = "pc_machine"
    from = "pc_nanotech"

    resources = {
        category = terraforming

        cost = {
            energy = 10000
        }
    }

    duration = 7200

    potential = {
        exists = owner
        has_ascension_perk = ap_machine_worlds
        has_tradition = tr_nanotech_3
    }

    ai_weight = {
        weight = 10
        modifier = {
            factor = 0
            from = {
                any_owned_species = {
                    NOR = {
                        has_trait = trait_machine_unit
                        has_trait = trait_mechanical
                    }
                    is_sapient = yes
                }
            }
        }
    }
}
