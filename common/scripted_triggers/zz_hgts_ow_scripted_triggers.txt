is_farming_district_uncapped = {
    OR = {
        uses_district_set = shattered_ring_world
        AND = {
            exists = owner_or_space_owner
            owner_or_space_owner = { is_anglers_empire = yes }
            is_planet_class = pc_ocean
        }
        is_planet_class = pc_gaia
        has_planet_flag = uncapped_farming_districts 			# For Modders
    }
}

is_mining_district_uncapped = {
    OR = {
        uses_district_set = shattered_ring_world
        AND = {
            exists = owner_or_space_owner
            owner_or_space_owner = {
                is_subterranean_empire = yes
            }
        }
        is_planet_class = pc_gaia
        has_planet_flag = uncapped_mining_districts 			# For Modders
    }
}

is_generator_district_uncapped = {
    OR = {
        uses_district_set = shattered_ring_world
        is_planet_class = pc_gaia
        has_planet_flag = uncapped_generator_districts 			# For Modders
    }
}
