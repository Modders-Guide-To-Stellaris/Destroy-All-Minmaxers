hgts_set_jump_allow_allies = {
    potential = {
        always = yes
    }
    allow = {
        hidden_trigger = {
            NOT = {
                has_global_flag = hgts_jump_allow_allies
            }
        }
    }
    effect = {
        hidden_effect = {
            hgts_clear_jump_settings = yes
            set_global_flag = hgts_jump_allow_allies
        }
    }
}

hgts_set_jump_allow_always = {
    potential = {
        always = yes
    }
    allow = {
        hidden_trigger = {
            OR = {
                has_global_flag = hgts_jump_allow_allies
                has_global_flag = hgts_jump_allow_neutral
            }
        }
    }
    effect = {
        hidden_effect = {
            hgts_clear_jump_settings = yes
        }
    }
}
